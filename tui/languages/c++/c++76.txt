#include <iostream>
#include <thread>
#include <mutex>
std::mutex mtx;
void printSafe(int id) {
    mtx.lock();
    std::cout << "Thread " << id << '\n';
    mtx.unlock();
}
int main() {
    std::thread t1(printSafe, 1);
    std::thread t2(printSafe, 2);
    t1.join();
    t2.join();
    return 0;
}
